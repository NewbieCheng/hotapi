 export interface ApifoxModel { 
     /** 
      * 发布时间，不传默认不限 
      */ 
     filter_note_time?: FilterNoteTime; 
     /** 
      * 笔记类型 
      */ 
     filter_note_type?: FilterNoteType; 
     /** 
      * 关键词 
      */ 
     keyword: string; 
     /** 
      * 页数，从1开始 
      */ 
     page: number; 
     /** 
      * 搜索ID，第一次请求可不传，服务端会生成searchId，翻页时需要携带服务端返回的searchId 
      */ 
     search_id?: string; 
     /** 
      * 会话ID，第一次请求可不传，服务端会生成sessionId，翻页时携带服务端返回的sessionId 
      */ 
     session_id?: string; 
     /** 
      * 排序 
      */ 
     sort_type?: SortType; 
     [property: string]: any; 
 }

  * 发布时间，不传默认不限
